```python
def pig_latin(text):
  words = text.split()
  pig_latin_words = []
  for word in words:
    endString = word[1:].lower()
    firstChar = word[0].lower()
    pig_latin_words.append(endString + firstChar + 'ay')
  return ' '.join(pig_latin_words)
```
