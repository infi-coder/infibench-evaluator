1. Using `if let` with a condition:

    ```rust

        let mut foo: Option<u8> = None;
        if let Some(foo_val) = foo {
            if foo_val < 5 {
                foo.replace(5);
            }
        } else {
            foo.replace(5);
        }
    }
    ```

2. Using `match` with a guard:

    ```rust

        let mut foo: Option<u8> = None;
        match foo {

        }
    }
    ```

    fn main() {
