You can fill in the blanks in your JavaScript code as follows:

```javascript
document.getElementById('check').onclick = function() {
  let allAreFilled = true;
  document.getElementById("myForm").querySelectorAll('input,textarea,select').forEach(function(i) {
    if (!allAreFilled) return;
    if (i.type === "radio") {
      let radioValueCheck = false;
      document.getElementById("myForm").querySelectorAll(`input[name=${i.name}]`).forEach(function(r) {
        if (r.checked) radioValueCheck = true;
      })
      allAreFilled = radioValueCheck;
      return;
    }
    if (!i.value) { allAreFilled = false;  return; }
  })
  if (!allAreFilled) {
    alert('Fill all the fields');
  }
};
```

This code will check all the `input`, `textarea`, and `select` elements inside the `myForm` div when the button with id `check` is clicked. If any required field is not filled, it will show an alert saying 'Fill all the fields'.
