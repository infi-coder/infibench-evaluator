import string

    insert_position = random.randint(0, 10)
    random_string = random_string[:insert_position] + "NAA3U" + random_string[insert_position:]
    
    # Insert two random digits at random positions in the string
    for i in range(2):
        insert_position = random.randint(0, len(random_string) - 1)

        random_string = random_string[:insert_position] + random_digit + random_string[insert_position:]
    
    return random_string
